gost配置端口转发

项目地址：https://github.com/ginuerzh/gost

1.服务端

下载Linux客户端

wget https://github.com/ginuerzh/gost/releases/download/v2.5/gost_2.5_linux_amd64.tar.gz
tar -zxvf gost_2.5_linux_amd64.tar.gz
mv gost_2.5_linux_amd64 /usr/local/gost_2.5
cd /usr/local/gost_2.5

注册systemd服务
vim gost.service

[Unit]
Description=SS
After=syslog.target
After=network.target


[Service]
Type=simple
User=root
Group=root
WorkingDirectory=/usr/local/gost_2.5
ExecStart=/usr/local/gost_2.5/gost -L :8848 -L ss://aes-256-cfb:superman125@:8869
Restart=always


[Install]
WantedBy=multi-user.target

cp gost.service /etc/systemd/system
chmod 754 /etc/systemd/system/gost.service
systemctl enable gost.service
systemctl start gost.service

# centos 查看防火墙状态：firewall-cmd --state
# 列出防火墙配置：firewall-cmd --list-all
# 永久开放80端口：firewall-cmd --permanent --zone=public --add-port=8866/tcp
# 重载防火墙配置：firewall-cmd —-reload
# 直接启动socks5：screen -dmS SK5 ./gost  -L socks5://:3900s

######国内机器跳转翻墙######
# 在香港机器执行：gost -L=kcp://:8833
# 在国内机器执行：gost -L=socks5://:5511 -F socks5+kcp://x.x.x.x:8833

##########远程连接桌面##########
被控机A    ./gost -L=rtcp://:5900/localhost:5900 -F socks5+quic://xxxxxxx:8866

主控机B    ./gost -L=tcp://:8848/localhost:5900 -F socks5+quic://xxxxxxxx:8866    

服务器C    ./gost -L=quic://:8866
